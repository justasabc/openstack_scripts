# 2. pre-check

# 3. controller node
## 3.1 set env 

	vim ./same/interfaces
	source ./set_env

## 3.2 install 

	./controller/install

## 3.3 uninstall

	./controller/uninstall

## 3.4 reinstall nova

	./controller/nova_uninstall
	./controller/nova_install

## 4.1 set env 

	vim ./same/interfaces
	source ./set_env

## 4.2 install 

	./node/install

## 4.2 uninstall
